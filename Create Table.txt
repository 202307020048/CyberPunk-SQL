
-- Drop existing tables if they exist
DROP TABLE IF EXISTS DENIZEN;
DROP TABLE IF EXISTS ITEMS;
DROP TABLE IF EXISTS NEXUS;

-- Create DENIZEN Table
CREATE TABLE DENIZEN (
    netizen_id INTEGER PRIMARY KEY,
    name TEXT,
    nexus_id INTEGER,
    gender TEXT,
    role TEXT,
    credits INTEGER,
    status TEXT
);

-- Create ITEMS Table
CREATE TABLE ITEMS (
    name TEXT,
    owner TEXT
);

-- Create NEXUS Table
CREATE TABLE NEXUS (
    nexus_id INTEGER PRIMARY KEY,
    name TEXT,
    overseer INTEGER
);
